# Database Schema Documentation

## Core Tables

### 1. Users (Modified)

```sql
CREATE TABLE public."Users" (
  u_id UUID PRIMARY KEY,
  email TEXT NOT NULL UNIQUE,
  role TEXT NOT NULL DEFAULT 'STUDENT' CHECK (role IN ('STUDENT', 'PARENT', 'COLLEGE')),
  created_at TIMESTAMPTZ DEFAULT NOW()
);
```

### 2. StudentProfiles

```sql
CREATE TABLE public."StudentProfiles" (
  id UUID PRIMARY KEY,
  user_id UUID NOT NULL UNIQUE REFERENCES "Users"(u_id),
  full_name TEXT,
  gender TEXT,
  locality TEXT,
  category TEXT,
  budget INTEGER,
  extracurriculars TEXT[],
  hobbies TEXT[],
  importance_locality INTEGER,
  importance_financial INTEGER,
  importance_eligibility INTEGER,
  importance_events_hobbies INTEGER,
  importance_quality INTEGER,
  is_profile_complete BOOLEAN,
  invite_code TEXT UNIQUE,
  invite_code_expires_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ,
  updated_at TIMESTAMPTZ
);
```

### 3. ParentProfiles

```sql
CREATE TABLE public."ParentProfiles" (

```

### 4. CollegeProfiles

```sql
CREATE TABLE public."CollegeProfiles" (

  aishe_code TEXT,

  designation TEXT,
  contact_email TEXT,
```

### 5. ParentStudentLinks

```sql
CREATE TABLE public."ParentStudentLinks" (
  id UUID PRIMARY KEY,
  parent_id UUID REFERENCES "Users"(u_id),
  student_id UUID REFERENCES "Users"(u_id),
  status TEXT CHECK (status IN ('PENDING', 'ACCEPTED', 'REJECTED', 'REVOKED')),
  requested_at TIMESTAMPTZ,
  responded_at TIMESTAMPTZ,
  UNIQUE(parent_id, student_id)
);
```

### 6. QuizResults

```sql
CREATE TABLE public."QuizResults" (
  id UUID PRIMARY KEY,
  user_id UUID REFERENCES "Users"(u_id),
  score_logical FLOAT,
  score_quant FLOAT,
  score_analytical FLOAT,
  score_verbal FLOAT,

  riasec_artistic FLOAT,

  riasec_enterprising FLOAT,

  created_at TIMESTAMPTZ,

```

### 7. Recommendations

```sql
CREATE TABLE public."Recommendations" (
```

### 8. Notifications

```sql
CREATE TABLE public."Notifications" (
```

### RIASEC

```sql
CREATE TABLE public."RIASEC" (
  "Title" TEXT NOT NULL PRIMARY KEY,
  "Realistic" DOUBLE PRECISION,
  "Investigative" DOUBLE PRECISION,
  "Artistic" DOUBLE PRECISION,
  "Social" DOUBLE PRECISION,
  "Enterprising" DOUBLE PRECISION,
  "Conventional" DOUBLE PRECISION,
  "desc" TEXT
);
```

### career_to_course

```sql
CREATE TABLE public.career_to_course (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY,
  careers TEXT REFERENCES "RIASEC"("Title"),
  courses TEXT REFERENCES "Courses"("Courses")
);
```

### course_to_college

```sql
CREATE TABLE public.course_to_college (
  id BIGINT PRIMARY KEY,
  courses TEXT REFERENCES "Courses"("Courses"),
  colleges TEXT REFERENCES "CollegeList"("Name")
);
```
